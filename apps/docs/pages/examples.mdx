## Card Layout

A responsive card layout showcasing different component compositions:

<LiveExample
  code={`import { Box, Button, Text } from '@beauginbey/vanilla-components';

export default function CardGrid() {
  const cards = [
    { title: 'Design', desc: 'Create beautiful interfaces', color: 'info' },
    { title: 'Develop', desc: 'Build with modern tools', color: 'success' },
    { title: 'Deploy', desc: 'Ship with confidence', color: 'brand' },
  ];

  return (
    <Box 
      display="grid" 
      style={{ 
        gridTemplateColumns: 'repeat(auto-fit, minmax(250px, 1fr))',
        gap: 'var(--spacing-4)'
      }}
    >
      {cards.map((card) => (
        <Box
          key={card.title}
          p={5}
          backgroundColor="primary"
          borderRadius="lg"
          boxShadow="base"
          style={{
            transition: 'transform 0.2s, box-shadow 0.2s',
            cursor: 'pointer',
          }}
          onMouseEnter={(e) => {
            e.currentTarget.style.transform = 'translateY(-4px)';
            e.currentTarget.style.boxShadow = 'var(--shadow-lg)';
          }}
          onMouseLeave={(e) => {
            e.currentTarget.style.transform = 'translateY(0)';
            e.currentTarget.style.boxShadow = 'var(--shadow-base)';
          }}
        >
          <Text 
            as="h3" 
            size="xl" 
            weight="bold" 
            mb={2}
            color={card.color as any}
          >
            {card.title}
          </Text>
          <Text color="secondary" mb={3}>
            {card.desc}
          </Text>
          <Button variant="outline" size="sm">
            Learn more ‚Üí
          </Button>
        </Box>
      ))}
    </Box>
  );
}`}
/>

## Alert Messages

Different types of alert components:

<LiveExample
  code={`import { Box, Text } from '@beauginbey/vanilla-components';

export default function Alerts() {
  const alerts = [
    { type: 'success', message: 'Your changes have been saved successfully!' },
    { type: 'warning', message: 'Please review your input before submitting.' },
    { type: 'error', message: 'An error occurred. Please try again.' },
    { type: 'info', message: 'New features are now available in your account.' },
  ];

  return (
    <Box display="flex" flexDirection="column" gap={3}>
      {alerts.map((alert) => (
        <Box
          key={alert.type}
          p={3}
          borderRadius="md"
          display="flex"
          alignItems="center"
          gap={2}
          style={{
            backgroundColor: \`var(--color-feedback-\${alert.type})\`,
            color: 'white',
            opacity: 0.9,
          }}
        >
          <Text weight="semibold">
            {alert.type.charAt(0).toUpperCase() + alert.type.slice(1)}:
          </Text>
          <Text>{alert.message}</Text>
        </Box>
      ))}
    </Box>
  );
}`}
/>

export default function PricingCards() {
  const plans = [
    { name: 'Starter', price: '$9', features: ['1 User', '10 Projects', 'Basic Support'] },

    { name: 'Enterprise', price: 'Custom', features: ['Unlimited Users', 'Custom Features', '24/7 Support'] },
  ];

  return (
    <Box
      display="grid"
      style={{
        gridTemplateColumns: 'repeat(auto-fit, minmax(280px, 1fr))',
        gap: 'var(--spacing-4)',

    >
      {plans.map((plan) => (
        <Box
          key={plan.name}
          p={6}
          backgroundColor={plan.popular ? 'brand' : 'secondary'}
          color={plan.popular ? 'inverse' : 'primary'}
          borderRadius="xl"
          textAlign="center"
          position="relative"
          style={{
            transform: plan.popular ? 'scale(1.05)' : 'scale(1)',
          }}
        >
          {plan.popular && (
            <Box
              position="absolute"
              top={-12}
              left="50%"
              style={{ transform: 'translateX(-50%)' }}

              px={3}

            {plan.price !== 'Custom' && <Text as="span" size="base" weight="normal">/mo</Text>}
